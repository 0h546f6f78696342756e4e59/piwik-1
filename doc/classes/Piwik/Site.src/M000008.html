    def destroy
      raise UnknownSite, "Site not existent in Piwik yet, call 'save' first" if new?
      xml = call('SitesManager.deleteSite', :idSite => id)
      result = XmlSimple.xml_in(xml, {'ForceArray' => false})
      freeze
      result['success'] ? true : false
    end
